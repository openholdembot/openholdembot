Most important thing ahead:

- PRESS 'q' to quit :) (because this will always move mouse button to top left corner to avoid any button color changes)
- you have to use "windows classic theme", "XP theme" is not supported atm

This script must be executed on the same machine the titan poker client is installed on, ofc if you want to be 100% safe you will have to use some kind of stealth method. Ofc you can always create a fake account and make a testrun at play money tables.

It will completely manage the poker client with the specified options, however i would not recommend using this for NL, Replica Breitling Watches because there is no rebuy function thus far.
But be aware that non poker client related stuff (OH instances/bring etc) isn't managed here, i excluded it because you really don't want to run OH on the poker client machine.

You will need to install Autoit, because i did not make sort of config for this but simply made a "options.au3" include, it's wise anyway to look through the source code of any au3 script you did not create yourself before executing it...

After setting the options in "options.au3" simple compile "TitanManager.au3" and you are ready to go.

All 3 files are downloadable in the Files section.

Func StartClient()
	
	Local $need_login = 0
	
	While ProcessExists($PCExe) = 0
		Run($PCDir & $PCExe, $PCDir)
		$need_login = 1
		Sleep(3000)
	WEnd
		
	If $need_login = 1 Then
		
		WinWait("Titan Poker login","", 10)
		Sleep(3000)
	
		Login()
	
	EndIf

EndFunc

Func CloseClient()
	While ProcessExists($PCExe)
		ProcessClose($PCExe)
		;Wait 5 min
		Sleep(5*60*1000)
	WEnd
EndFunc

Func Count_Tables()
	
	$returnvar = 0
	
	$var3 = WinList()

	For $u = 1 to $var3[0][0]
		If StringInStr($var3[$u][0], $Client_Table_Keyword) > 0 And StringInStr($var3[$u][0], $OH_Inst_Keyword) = 0 Then
			$returnvar += 1
		EndIf
	Next
	
	return $returnvar
	
EndFunc

Func ManageAll()
	
	;got disconnected need to login
	If WinExists("Titan Poker login") And IsVisible(WinGetHandle("Titan Poker login")) Then
		Login()
	EndIf
	
	;Move Mouse away to avoid buttons colors changes
	MouseMove(0,0,0)
	
	;count the amount of tables
	$Current_TInstances = Count_Tables()
	
	;While not max tables reached add more
	If $Current_TInstances < $Max_TInstances Then
		;If not already on WL then add to WL
		If $WLOnList = 0 And PixelGetColor(641+$LobbyX, 541+$LobbyY) <> $LRemoveWL Then WLAdd($LobbyX, $LobbyY)
		;Somtimes WL does not work -> retry after 180 sec
		If Mod(@SEC + @MIN*60, 180) = 0 Then 
			$WLOnList = 0
			WLAdd($LobbyX, $LobbyY)
		EndIf
	EndIf	

	;List of all running wins
	$var = WinList()

	For $i = 1 to $var[0][0]
		
		$winsize = WinGetClientSize($var[$i][1])
		$winpos = WinGetPos($var[$i][1])
		
		If @error = 0 Then
			
			;disconnected popup
			If $var[$i][0] = "Titan Poker" Then
				;Cannot connect to Poker server -> close client and restart
				If $winsize[0] = 402 And $winsize[1] = 265 And IsVisible($var[$i][1]) Then
					CloseClient()
					Restart()
				EndIf
			EndIf	
			
			;confirm leaving table popup
			If $var[$i][0] = "Confirm leaving tournament" Then
				If $winsize[0] = 402 And $winsize[1] = 265 And IsVisible($var[$i][1]) Then
					WinActivate($var[$i][1])
					MouseClick("left", 240+$winpos[0], 240+$winpos[1], 1, 0)
				EndIf
			EndIf
			
			;manage popups before poker client table manage
			If $var[$i][0] = "" Then		
				;Promotion popup during play
				If $winsize[0] = 402 And $winsize[1] = 400 And IsVisible($var[$i][1]) Then

					EndIf
					;Table Popup: Buyin
					If PixelGetColor(28+$winpos[0], 56+$winpos[1]) = 0xFFFFFF And PixelGetColor(132+$winpos[0], 63+$winpos[1]) = 0xFFFFFF And PixelGetColor(371+$winpos[0], 53+$winpos[1]) = 0xFFFFFF Then
						WinActivate($var[$i][1])
						MouseClick("left", 42+$winpos[0], 203+$winpos[1], 1, 0)
						Sleep(500)
						MouseClick("left", 160+$winpos[0], 203+$winpos[1], 1, 0)
						Send("{DEL}")
						Send("{DEL}")
						Send("{DEL}")
						Send("{DEL}")
						Send("{DEL}")
						Send($buyin)
						MouseClick("left", 158+$winpos[0], 245+$winpos[1], 1, 0)
					EndIf
				EndIf
			EndIf
			;manage poker client table
			If StringInStr($var[$i][0], $Client_Table_Keyword) > 0 Then
				If $winsize[0] = 512 And $winsize[1] = 324 And IsVisible($var[$i][1]) Then
					;Not playing at table and Join WL is visible -> close table
					If PixelGetColor(256+$winpos[0], 297+$winpos[1]) = $TJWLPC1 And PixelGetColor(277+$winpos[0], 299+$winpos[1]) = $TJWLPC2 And PixelGetColor(290+$winpos[0], 314+$winpos[1]) = $TJWLPC3 Then
						WinClose( $var[$i][1] )
					EndIf	
					;T check if i am back button visible
					If PixelGetColor(258+$winpos[0], 305+$winpos[1]) = $TIamback1PC And PixelGetColor(271+$winpos[0], 306+$winpos[1]) = $TIamback2PC And PixelGetColor(309+$winpos[0], 311+$winpos[1]) = $TIamback3PC Then
						WinActivate($var[$i][1])
						MouseClick("left", 280+$winpos[0], 310+$winpos[1], 1, 0)
						Sleep(500)
					EndIf					
					;T check if post or Wait for BB
					If PixelGetColor(306+$winpos[0], 314+$winpos[1]) = $TWaitBB1PC And PixelGetColor(355+$winpos[0], 314+$winpos[1]) = $TWaitBB2PC And PixelGetColor(421+$winpos[0], 320+$winpos[1]) = $TWaitBB3PC Then					
						WinActivate($var[$i][1])
						MouseClick("left", 330+$winpos[0], 320+$winpos[1], 1, 0)
						Sleep(500)
					EndIf
					;T check if post SB/BB or sitout
					If PixelGetColor(325+$winpos[0], 316+$winpos[1]) = $TpostSO1PC And PixelGetColor(332+$winpos[0], 323+$winpos[1]) = $TpostSO2PC And PixelGetColor(363+$winpos[0], 318+$winpos[1]) = $TpostSO3PC Then
						;check if enough players are still playing
						If TPlaying($winpos[0], $winpos[1]) < $minplayer Then
							;leave if not
							TLeave($winpos[0], $winpos[1])
						Else
							WinActivate($var[$i][1])
							MouseClick("left", 227+$winpos[0], 318+$winpos[1], 1, 0)
							Sleep(500)
						EndIf
					EndIf
				EndIf
			EndIf
		EndIf
		
	Next
	
EndFunc

Func KillStartPopUp()
	
	;List of all running wins
	$var = WinList()

	For $i = 1 to $var[0][0]
		
		$winsize = WinGetClientSize($var[$i][1])
		$winpos = WinGetPos($var[$i][1])
		
		If @error = 0 Then
			;manage popups before poker client table manage
			If $var[$i][0] = "" Then
				;Promotion popup at start of client
				If $winsize[0] = 402 And $winsize[1] = 265 And IsVisible($var[$i][1]) And PixelGetColor($winpos[0]+364, $winpos[1]+249) = $LStartPopup Then
					WinActivate($var[$i][1])
					MouseClick("left", 360+$winpos[0], 251+$winpos[1], 1, 0)
				EndIf				
			EndIf
		EndIf
		
	Next
	
EndFunc

Func Restart()

	StartClient()
	Sleep(10000)
	
	KillStartPopUp()

	QS_dolist($LobbyX, $LobbyY)

	Sleep(3000)

EndFunc
